\[\frac{V_{in}(s) - V_{out}(s)}{R} = sCV_{out}(s) + \frac{1}{sL}V_{out}(s)\]
\[V_{in}(s)(\frac{1}{R}) = V_{out}(s)(\frac{1}{R} + sC + \frac{1}{sL})\]
